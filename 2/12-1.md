# 一法通用 · 函数模板

**模板**是 C++ 泛型编程的基础。函数模板让我们能够编写与类型无关的代码，由编译器根据实际类型生成具体的函数。

------

**为什么需要模板**

```cpp
// 没有模板，需要为每种类型写一个函数
int maxInt(int a, int b) { return a > b ? a : b; }
double maxDouble(double a, double b) { return a > b ? a : b; }
string maxString(string a, string b) { return a > b ? a : b; }

// 有了模板，一个函数就够了
template<typename T>
T maximum(T a, T b)
{
    return a > b ? a : b;
}

int main()
{
    cout << maximum(3, 5) << endl;        // int
    cout << maximum(3.14, 2.72) << endl;  // double
    cout << maximum("abc"s, "def"s) << endl;  // string
    
    return 0;
}
```

------

**模板语法**

```cpp
// template 关键字 + 模板参数列表
template<typename T>  // 或 template<class T>
T add(T a, T b)
{
    return a + b;
}

// 多个模板参数
template<typename T, typename U>
auto add(T a, U b) -> decltype(a + b)
{
    return a + b;
}

// C++14 简化
template<typename T, typename U>
auto add14(T a, U b)
{
    return a + b;
}

int main()
{
    cout << add(1, 2) << endl;        // 3
    cout << add(1.5, 2.5) << endl;    // 4.0
    cout << add(1, 2.5) << endl;      // 3.5
    
    return 0;
}
```

------

**模板实例化**

```cpp
template<typename T>
void print(T value)
{
    cout << value << endl;
}

int main()
{
    // 隐式实例化：编译器推断类型
    print(42);        // T = int
    print("hello");   // T = const char*
    
    // 显式实例化
    print<double>(42);  // T = double
    
    return 0;
}
```

------

**非类型模板参数**

```cpp
// N 是非类型参数
template<typename T, int N>
class Array
{
    T data[N];
public:
    int size() const { return N; }
    T& operator[](int i) { return data[i]; }
};

// 函数模板也可以
template<int N>
int power(int base)
{
    int result = 1;
    for (int i = 0; i < N; ++i)
        result *= base;
    return result;
}

int main()
{
    Array<int, 10> arr;
    cout << arr.size() << endl;  // 10
    
    cout << power<3>(2) << endl;  // 8 (2^3)
    
    return 0;
}
```

------

**模板特化**

```cpp
// 通用版本
template<typename T>
void print(T value)
{
    cout << "Generic: " << value << endl;
}

// 完全特化：针对特定类型
template<>
void print<bool>(bool value)
{
    cout << "Bool: " << (value ? "true" : "false") << endl;
}

// 指针的部分特化（只有类模板支持）
template<typename T>
void print(T* ptr)
{
    if (ptr)
        cout << "Pointer to: " << *ptr << endl;
    else
        cout << "Null pointer" << endl;
}

int main()
{
    print(42);        // Generic: 42
    print(true);      // Bool: true
    
    int x = 100;
    print(&x);        // Pointer to: 100
    
    return 0;
}
```

**SFINAE 与 enable_if**

```cpp
#include <type_traits>

// 只对整数类型启用
template<typename T>
typename enable_if<is_integral<T>::value, T>::type
doubleValue(T x)
{
    cout << "Integer version" << endl;
    return x * 2;
}

// 只对浮点类型启用
template<typename T>
typename enable_if<is_floating_point<T>::value, T>::type
doubleValue(T x)
{
    cout << "Floating point version" << endl;
    return x * 2.0;
}

int main()
{
    cout << doubleValue(5) << endl;      // Integer version: 10
    cout << doubleValue(3.14) << endl;   // Floating point version: 6.28
    
    return 0;
}
```

**可变参数模板**

```cpp
// C++11 可变参数模板
// 基础情况
void print()
{
    cout << endl;
}

// 递归情况
template<typename T, typename... Args>
void print(T first, Args... rest)
{
    cout << first << " ";
    print(rest...);  // 递归展开
}

// C++17 折叠表达式
template<typename... Args>
void print17(Args... args)
{
    ((cout << args << " "), ...);
    cout << endl;
}

int main()
{
    print(1, 2.0, "three", '4');
    print17(1, 2.0, "three", '4');
    
    return 0;
}
```

------

**完美转发**

```cpp
#include <utility>

void process(int& x) { cout << "Lvalue: " << x << endl; }
void process(int&& x) { cout << "Rvalue: " << x << endl; }

// 完美转发：保持参数的值类别
template<typename T>
void wrapper(T&& arg)
{
    process(forward<T>(arg));
}

int main()
{
    int x = 42;
    wrapper(x);    // Lvalue: 42
    wrapper(100);  // Rvalue: 100
    
    return 0;
}
```

------

**constexpr 函数模板**

```cpp
template<typename T>
constexpr T factorial(T n)
{
    return n <= 1 ? 1 : n * factorial(n - 1);
}

int main()
{
    constexpr int f5 = factorial(5);  // 编译期计算
    cout << f5 << endl;  // 120
    
    int n = 6;
    cout << factorial(n) << endl;  // 运行期计算：720
    
    return 0;
}
```
