# 编译期魔法 · 模板元编程

模板元编程是利用模板在编译期进行计算的技术。虽然复杂，但能带来零运行时开销的抽象。

------

**编译期计算**

```cpp
// 编译期阶乘
template<int N>
struct Factorial
{
    static const int value = N * Factorial<N - 1>::value;
};

// 特化作为递归终止条件
template<>
struct Factorial<0>
{
    static const int value = 1;
};

int main()
{
    // 编译期计算，运行期直接使用结果
    cout << Factorial<5>::value << endl;   // 120
    cout << Factorial<10>::value << endl;  // 3628800
    
    // 可以用作数组大小
    int arr[Factorial<4>::value];  // int arr[24]
    
    return 0;
}
```

template<typename T>
struct RemovePointer
{
    using type = T;
};

template<typename T>
struct RemovePointer<T*>
{
    using type = T;
};

// 使用
int main()
{
    RemovePointer<int*>::type a = 42;      // int a
    RemovePointer<double**>::type b = nullptr;  // double* b
    
    cout << typeid(a).name() << endl;  // int
    
    return 0;
}

**类型列表**

```cpp
// 类型列表
template<typename... Types>
struct TypeList {};

// 获取第一个类型
template<typename List>
struct Front;

template<typename Head, typename... Tail>
struct Front<TypeList<Head, Tail...>>
{
    using type = Head;
};

// 获取大小
template<typename List>
struct Size;

template<typename... Types>
struct Size<TypeList<Types...>>
{
    static const size_t value = sizeof...(Types);
};

int main()
{
    using MyList = TypeList<int, double, string>;
    
    Front<MyList>::type x = 42;  // int x
    cout << Size<MyList>::value << endl;  // 3
    
    return 0;
}
```

**SFINAE 进阶**

```cpp
// 检测类型是否有特定成员
template<typename T, typename = void>
struct HasSize : false_type {};

template<typename T>
struct HasSize<T, void_t<decltype(declval<T>().size())>> : true_type {};

// 使用
template<typename T>
void printInfo(const T& x)
{
    if constexpr (HasSize<T>::value)
    {
        cout << "Size: " << x.size() << endl;
    }
    else
    {
        cout << "No size method" << endl;
    }
}

int main()
{
    vector<int> v = {1, 2, 3};
    int n = 42;
    
    printInfo(v);  // Size: 3
    printInfo(n);  // No size method
    
    return 0;
}
```

**编译期字符串**

```cpp
// 固定大小的编译期字符串
template<size_t N>
struct FixedString
{
    char data[N];
    
    constexpr FixedString(const char (&str)[N])
    {
        for (size_t i = 0; i < N; ++i)
            data[i] = str[i];
    }
    
    constexpr size_t size() const { return N - 1; }
    constexpr char operator[](size_t i) const { return data[i]; }
};

template<FixedString Str>
struct StringWrapper
{
    static void print()
    {
        cout << Str.data << endl;
    }
};

int main()
{
    StringWrapper<"Hello">::print();  // Hello (C++20)
    return 0;
}
```

**constexpr if（C++17）**

```cpp
template<typename T>
string stringify(const T& x)
{
    if constexpr (is_same_v<T, string>)
    {
        return x;  // 已经是字符串
    }
    else if constexpr (is_arithmetic_v<T>)
    {
        return to_string(x);  // 数字转字符串
    }
    else if constexpr (is_pointer_v<T>)
    {
        if (x == nullptr)
            return "null";
        return stringify(*x);  // 递归解引用
    }
    else
    {
        return "unknown";
    }
}

int main()
{
    cout << stringify(42) << endl;        // 42
    cout << stringify(3.14) << endl;      // 3.140000
    cout << stringify("hello"s) << endl;  // hello
    
    int n = 100;
    cout << stringify(&n) << endl;        // 100
    
    return 0;
}
```

**concept（C++20）**

```cpp
// 定义概念
template<typename T>
concept Numeric = is_arithmetic_v<T>;

template<typename T>
concept Printable = requires(T t) {
    { cout << t } -> same_as<ostream&>;
};

// 使用概念约束模板
template<Numeric T>
T add(T a, T b)
{
    return a + b;
}

template<Printable T>
void print(const T& x)
{
    cout << x << endl;
}

int main()
{
    cout << add(1, 2) << endl;      // OK
    cout << add(1.5, 2.5) << endl;  // OK
    // add("a", "b");  // 错误：string 不是 Numeric
    
    print(42);      // OK
    print("hello"); // OK
    
    return 0;
}
```

{
    return Unit<M1+M2, L1+L2, T1+T2>(a.value * b.value);
}

int main()
{
    Mass m(10);      // 10 kg

    // 编译期类型检查
    
    return 0;
}
