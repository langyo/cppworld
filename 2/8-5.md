# 电报机 · 字符串流

有时候，我们需要在字符串和其他数据类型之间进行转换，或者需要对字符串进行格式化处理。**字符串流**就是为此设计的工具——它让我们能够像操作文件一样操作字符串。

------

**字符串流类**

| 类 | 说明 |
| -- | ---- |
| `istringstream` | 从字符串读取数据 |
| `ostringstream` | 向字符串写入数据 |
| `stringstream` | 双向字符串流 |

使用前需要包含 `<sstream>` 头文件。

------

**从字符串读取数据**

```cpp
#include <sstream>
#include <string>
#include <iostream>
using namespace std;

int main()
{
    string data = "42 3.14 Hello";
    istringstream iss(data);
    
    int n;
    double d;
    string s;
    
    iss >> n >> d >> s;
    
    cout << "整数: " << n << endl;     // 42
    cout << "浮点数: " << d << endl;   // 3.14
    cout << "字符串: " << s << endl;   // Hello
    
    return 0;
}
```

------

**向字符串写入数据**

```cpp
#include <sstream>
#include <string>
#include <iostream>
using namespace std;

int main()
{
    ostringstream oss;
    
    oss << "The answer is " << 42;
    oss << " and pi is " << 3.14159;
    
    string result = oss.str();  // 获取字符串
    cout << result << endl;
    // The answer is 42 and pi is 3.14159
    
    return 0;
}
```

**字符串分割**

```cpp
#include <sstream>
#include <string>
#include <vector>
#include <iostream>
using namespace std;

vector<string> split(const string& s, char delimiter)
{
    vector<string> tokens;
    istringstream iss(s);
    string token;
    
    while (getline(iss, token, delimiter))
    {
        tokens.push_back(token);
    }
    
    return tokens;
}

int main()
{
    string csv = "apple,banana,cherry,date";
    auto parts = split(csv, ',');
    
    for (const auto& p : parts)
    {
        cout << p << endl;
    }
    // apple
    // banana
    // cherry
    // date
    
    return 0;
}
```

**类型转换**

```cpp
#include <sstream>
#include <string>

// 任意类型转字符串
template<typename T>
string toString(const T& value)
{
    ostringstream oss;
    oss << value;
    return oss.str();
}

// 字符串转任意类型
template<typename T>
T fromString(const string& s)
{
    istringstream iss(s);
    T value;
    iss >> value;
    return value;
}

int main()
{
    // 数字转字符串
    string s1 = toString(42);
    string s2 = toString(3.14);
    
    // 字符串转数字
    int n = fromString<int>("123");
    double d = fromString<double>("3.14159");
    
    // 当然，C++11 提供了更简单的方法
    string s = to_string(42);
    int n2 = stoi("123");
    double d2 = stod("3.14");
    
    return 0;
}
```

**格式化输出**

```cpp
#include <sstream>
#include <iomanip>
#include <iostream>
using namespace std;

int main()
{
    ostringstream oss;
    
    // 设置宽度和填充
    oss << setw(10) << setfill('0') << 42;
    cout << oss.str() << endl;  // 0000000042
    
    oss.str("");  // 清空
    
    // 设置精度
    oss << fixed << setprecision(2) << 3.14159;
    cout << oss.str() << endl;  // 3.14
    
    oss.str("");
    
    // 十六进制
    oss << hex << 255;
    cout << oss.str() << endl;  // ff
    
    return 0;
}
```

**实际应用：解析 CSV**

```cpp
#include <sstream>
#include <string>
#include <vector>
#include <iostream>
using namespace std;

struct Person
{
    string name;
    int age;
    string city;
};

Person parseLine(const string& line)
{
    Person p;
    istringstream iss(line);
    
    getline(iss, p.name, ',');
    
    string ageStr;
    getline(iss, ageStr, ',');
    p.age = stoi(ageStr);
    
    getline(iss, p.city, ',');
    
    return p;
}

int main()
{
    vector<string> lines = {
        "Alice,25,New York",
        "Bob,30,Los Angeles",
        "Carol,28,Chicago"
    };
    
    for (const auto& line : lines)
    {
        Person p = parseLine(line);
        cout << p.name << " (" << p.age << ") - " << p.city << endl;
    }
    
    return 0;
}
```

**构建复杂字符串**

```cpp
#include <sstream>
#include <vector>

string buildSQL(const string& table, const vector<string>& columns,
                const string& condition)
{
    ostringstream oss;
    
    oss << "SELECT ";
    
    for (size_t i = 0; i < columns.size(); ++i)
    {
        if (i > 0) oss << ", ";
        oss << columns[i];
    }
    
    oss << " FROM " << table;
    
    if (!condition.empty())
    {
        oss << " WHERE " << condition;
    }
    
    return oss.str();
}

int main()
{
    vector<string> cols = {"name", "age", "email"};
    string sql = buildSQL("users", cols, "age > 18");
    cout << sql << endl;
    // SELECT name, age, email FROM users WHERE age > 18
    
    return 0;
}
```
